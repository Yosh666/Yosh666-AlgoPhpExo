#------------------------------------------------------------
# Table: ORIGINE
#------------------------------------------------------------

CREATE TABLE ORIGINE(
        ID_ORIGINE Int  Auto_increment  NOT NULL ,
        NOM_PAYS   Varchar (20) NOT NULL
	,CONSTRAINT ORIGINE_PK PRIMARY KEY (ID_ORIGINE)
)ENGINE=InnoDB;


#------------------------------------------------------------
# Table: MARQUE
#------------------------------------------------------------

CREATE TABLE MARQUE(
        ID_MARQUE  Int  Auto_increment  NOT NULL ,
        NOM_MARQUE Varchar (20) NOT NULL ,
        ID_ORIGINE Int NOT NULL
	,CONSTRAINT MARQUE_PK PRIMARY KEY (ID_MARQUE)

	,CONSTRAINT MARQUE_ORIGINE_FK FOREIGN KEY (ID_ORIGINE) REFERENCES ORIGINE(ID_ORIGINE)
)ENGINE=InnoDB;


#------------------------------------------------------------
# Table: COULEURS
#------------------------------------------------------------

CREATE TABLE COULEURS(
        ID_COULEUR Int  Auto_increment  NOT NULL ,
        COULEUR    Varchar (20) NOT NULL
	,CONSTRAINT COULEURS_PK PRIMARY KEY (ID_COULEUR)
)ENGINE=InnoDB;


#------------------------------------------------------------
# Table: MOTORISATION
#------------------------------------------------------------

CREATE TABLE MOTORISATION(
        ID_MOTEUR   Int  Auto_increment  NOT NULL ,
        TYPE_MOTEUR Varchar (20) NOT NULL
	,CONSTRAINT MOTORISATION_PK PRIMARY KEY (ID_MOTEUR)
)ENGINE=InnoDB;


#------------------------------------------------------------
# Table: MODELE
#------------------------------------------------------------

CREATE TABLE MODELE(
        ID_MODELE  Int  Auto_increment  NOT NULL ,
        NOM_MODELE Varchar (20) NOT NULL ,
        ID_MARQUE  Int NOT NULL
	,CONSTRAINT MODELE_PK PRIMARY KEY (ID_MODELE)

	,CONSTRAINT MODELE_MARQUE_FK FOREIGN KEY (ID_MARQUE) REFERENCES MARQUE(ID_MARQUE)
)ENGINE=InnoDB;


#------------------------------------------------------------
# Table: VEHICULE
#------------------------------------------------------------

CREATE TABLE VEHICULE(
        ID_VOITURE Int  Auto_increment  NOT NULL ,
        IMMAT      Varchar (20) NOT NULL ,
        NB_PORTES  Int NOT NULL ,
        ID_MODELE  Int NOT NULL ,
        ID_MOTEUR  Int NOT NULL
	,CONSTRAINT VEHICULE_PK PRIMARY KEY (ID_VOITURE)

	,CONSTRAINT VEHICULE_MODELE_FK FOREIGN KEY (ID_MODELE) REFERENCES MODELE(ID_MODELE)
	,CONSTRAINT VEHICULE_MOTORISATION0_FK FOREIGN KEY (ID_MOTEUR) REFERENCES MOTORISATION(ID_MOTEUR)
)ENGINE=InnoDB;


#------------------------------------------------------------
# Table: colorer
#------------------------------------------------------------

CREATE TABLE colorer(
        ID_VOITURE Int NOT NULL ,
        ID_COULEUR Int NOT NULL
	,CONSTRAINT colorer_PK PRIMARY KEY (ID_VOITURE,ID_COULEUR)

	,CONSTRAINT colorer_VEHICULE_FK FOREIGN KEY (ID_VOITURE) REFERENCES VEHICULE(ID_VOITURE)
	,CONSTRAINT colorer_COULEURS0_FK FOREIGN KEY (ID_COULEUR) REFERENCES COULEURS(ID_COULEUR)
)ENGINE=InnoDB;
